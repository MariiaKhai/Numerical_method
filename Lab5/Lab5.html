def k_range(k, n):
    for i in range(n):
        if i != k:
            yield i

def L(x, y, xx):
    n, yy = len(x), 0
    for k in range(n):
        li = 1
        for j in k_range(k, n):
            li *= (xx - x[j]) / (x[k] - x[j])
        yy += li * y[k]
    return yy
